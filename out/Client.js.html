const EventEmitter = require('events')
const DataStore = require('../util/DataStore')
const { UserStore, GuildStore, ChannelStore, InteractionManager, RoleStore } = require('../util/Stores')
const login = require('../login')
const hb = require('../gateway/heartbeat')
const RESTManager = require('./rest/RESTManager')

const ClientMonitor = require('./Monitor')

import WebSocket from 'ws'

/**
 * @typedef
 */
class Client extends EventEmitter {
	constructor(options={intents: 32767}) {
		super()
		
		/**
		 * @type {Object}
		 */
		this.options = options;
		
		this.voice;
		
		try {
			this.voice = require('@discordjs/voice')
		} catch(err) {
			this.emit("debug", `@discordjs/voice not found, so voice support is disabled`)

			this.voice = null;
		}

		this.user = null;
		/**
		 * @type {UserStore}
		 */
		this.users = new UserStore(this)
		/**
		 * @type {GuildStore}
		 */
		this.guilds = new GuildStore(this)
		/**
		 * @type {ChannelStore}
		 */
		this.channels = new ChannelStore(this)
		/**
		 * @type {RoleStore}
		 */
		this.roles = new RoleStore(this)
		
		/**
		 * @type {String}
		 */
		this.token = options.token
		/**
		 * @type {Number}
		 */
		this.intents = options.intents || 32767

		/**
		 * @type {ClientMonitor}
		 */
		this.monitor = new ClientMonitor(this);

		/**
		 * @type {RESTManager}
		 */
		this.rest = new RESTManager(this)

		/**
		 * @type {DataStore}
		 */
		this.commands = new DataStore()

		/**
		 * @type {Boolean}
		 */
		this.readyAt;

		/**
		 * @type {Object}
		 */
		this.ws = {
			/**
			 * @type {String}
			 */
			sessionId: null,
			/**
			 * @type {WebSocket}
			 */
			socket: null,
			/**
			 * @type {Number}
			 */
			ping: 0,
			/**
			 * @type {Object}
			 */
			heartbeat: {
				/**
				 * @type {Number}
				 */
				interval: 0,
				/**
				 * @type {Boolean}
				 */
				recieved: false,
				/**
				 * @type {Number}
				 */
				last: 0,
				/**
				 * @type {Function}
				 */
				startHeartBeat: hb
			},
		}

	}
	/**
	 * Connects the Client to Discord
	 * @returns {Promise<Client>}
	 */
	connect() {
		return new Promise((resolve, reject) => {
			resolve(login(this))
		})
	}
}

module.exports = Client;
